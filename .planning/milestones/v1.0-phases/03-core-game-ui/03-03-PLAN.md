**GameGridTest.kt** — Canvas touch tests require `performTouchInput` with coordinates:

```kotlin
@RunWith(RobolectricTestRunner::class)
@Config(sdk = [31])
class GameGridTest {
    @get:Rule
    val composeRule = createComposeRule()

    private fun makeBoard(vararg values: Int): IntArray {
        val board = IntArray(81)
        values.forEachIndexed { i, v -> if (i < 81) board[i] = v }
        return board
    }

    @Test
    fun `tapping top-left cell dispatches index 0`() {
        var clicked: Int? = null
        composeRule.setContent {
            ThemeMMD {
                GameGrid(
                    modifier = Modifier.size(360.dp),  // fixed size for deterministic tap coords
                    board = IntArray(81),
                    solution = IntArray(81),
                    givenMask = BooleanArray(81),
                    selectedCellIndex = null,
                    pencilMarks = Array(81) { emptySet() },
                    onCellClick = { clicked = it }
                )
            }
        }
        // Cell 0 = row 0, col 0. At 360.dp grid, cellSize = 40.dp.
        // Center of cell 0 = (20.dp, 20.dp) in dp-space. Use fraction-based tap:
        composeRule.onRoot().performTouchInput {
            val cellW = width / 9f
            val cellH = height / 9f
            click(Offset(cellW * 0.5f, cellH * 0.5f))  // center of cell (0,0)
        }
        assertThat(clicked).isEqualTo(0)
    }

    @Test
    fun `tapping bottom-right cell dispatches index 80`() {
        var clicked: Int? = null
        composeRule.setContent {
            ThemeMMD {
                GameGrid(
                    modifier = Modifier.size(360.dp),
                    board = IntArray(81),
                    solution = IntArray(81),
                    givenMask = BooleanArray(81),
                    selectedCellIndex = null,
                    pencilMarks = Array(81) { emptySet() },
                    onCellClick = { clicked = it }
                )
            }
        }
        composeRule.onRoot().performTouchInput {
            val cellW = width / 9f
            val cellH = height / 9f
            click(Offset(cellW * 8.5f, cellH * 8.5f))  // center of cell (8,8)
        }
        assertThat(clicked).isEqualTo(80)
    }

    @Test
    fun `tapping row 1 col 2 dispatches index 11`() {
        var clicked: Int? = null
        composeRule.setContent {
            ThemeMMD {
                GameGrid(
                    modifier = Modifier.size(360.dp),
                    board = IntArray(81),
                    solution = IntArray(81),
                    givenMask = BooleanArray(81),
                    selectedCellIndex = null,
                    pencilMarks = Array(81) { emptySet() },
                    onCellClick = { clicked = it }
                )
            }
        }
        composeRule.onRoot().performTouchInput {
            val cellW = width / 9f
            val cellH = height / 9f
            click(Offset(cellW * 2.5f, cellH * 1.5f))  // col=2, row=1
        }
        assertThat(clicked).isEqualTo(11)  // 1*9 + 2 = 11
    }

    @Test
    fun `renders without crash on empty board`() {
        composeRule.setContent {
            ThemeMMD {
                GameGrid(
                    modifier = Modifier.size(360.dp),
                    board = IntArray(81),
                    solution = IntArray(81),
                    givenMask = BooleanArray(81),
                    selectedCellIndex = null,
                    pencilMarks = Array(81) { emptySet() },
                    onCellClick = {}
                )
            }
        }
        // No exception thrown = pass
    }
}
```
