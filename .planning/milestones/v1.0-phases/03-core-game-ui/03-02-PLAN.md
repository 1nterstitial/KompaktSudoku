Create `GameGrid.kt`. This file contains the `CellData` data class and the `GameGrid` composable. The grid is drawn entirely with a single `Canvas` — no Box composables per cell.

**CellData (stable wrapper for per-cell state):**
```kotlin
import androidx.compose.runtime.Stable

@Stable
data class CellData(
    val index: Int,
    val digit: Int,          // 0 = empty
    val isGiven: Boolean,
    val isSelected: Boolean,
    val isError: Boolean,    // non-given digit present that != solution; D-04
    val pencilMarks: Set<Int>
)
```

`isError` = `digit != 0 && !isGiven && digit != solution[index]`.

**GameGrid composable signature:**
```kotlin
@Composable
fun GameGrid(
    modifier: Modifier = Modifier,
    board: IntArray,
    solution: IntArray,
    givenMask: BooleanArray,
    selectedCellIndex: Int?,
    pencilMarks: Array<Set<Int>>,
    onCellClick: (Int) -> Unit
)
```

**Implementation requirements:**

1. Build `cells: List<CellData>` using `remember(board, givenMask, selectedCellIndex, pencilMarks)` — prevents recomputation when unrelated state changes.

2. Hoist `val textMeasurer = rememberTextMeasurer()` at the composable level (cannot be called inside DrawScope).

3. Define text styles at composable level (captured by Canvas lambda):
   - `digitStyleGiven`: fontSize = 20.sp, fontWeight = FontWeight.Bold, color = Color.Black
   - `digitStylePlayer`: fontSize = 20.sp, fontWeight = FontWeight.Normal, color = Color.Black
   - `digitStyleSelectedGiven`: fontSize = 20.sp, fontWeight = FontWeight.Bold, color = Color.White
   - `digitStyleSelectedPlayer`: fontSize = 20.sp, fontWeight = FontWeight.Normal, color = Color.White
   - `pencilStyle`: fontSize = 9.sp, fontWeight = FontWeight.Normal, color = Color.Black

4. Use `BoxWithConstraints(modifier = modifier)` to compute square grid size:
   ```kotlin
   val gridSizeDp = minOf(maxWidth, maxHeight)
   val gridSizePx = with(LocalDensity.current) { gridSizeDp.toPx() }
   val cellSizePx = gridSizePx / 9f
   ```

5. Attach `Modifier.pointerInput(Unit) { detectTapGestures { offset -> ... } }` to the Canvas for cell selection:
   ```kotlin
   val col = (offset.x / cellSizePx).toInt().coerceIn(0, 8)
   val row = (offset.y / cellSizePx).toInt().coerceIn(0, 8)
   onCellClick(row * 9 + col)
   ```

6. **Canvas drawing order** (inside the Canvas lambda, calling DrawScope extensions):
   a. Selected cell background — solid black rect for the selected cell only (D-02)
   b. Error cell indicator — 1dp inset border for cells where `isError = true` (D-04 recommendation)
   c. Digit text — for each cell with `digit != 0`, drawText using appropriate style
   d. Pencil mark mini-grid — for each empty cell with non-empty pencilMarks
   e. Grid lines — draw all 10 horizontal + 10 vertical lines; `i % 3 == 0` → thick (2.5dp), else thin (1dp)

   Grid lines are drawn LAST so they appear on top of all cell backgrounds — this prevents thick borders being partially overwritten by adjacent cell fills.

**DrawScope extension for selected cell fill:**
```kotlin
private fun DrawScope.drawSelectedCell(cell: CellData, cellSize: Float) {
    val row = cell.index / 9
    val col = cell.index % 9
    drawRect(
        color = Color.Black,
        topLeft = Offset(col * cellSize, row * cellSize),
        size = Size(cellSize, cellSize)
    )
}
```

**DrawScope extension for error indicator (D-04):**
```kotlin
private fun DrawScope.drawErrorIndicator(cell: CellData, cellSize: Float) {
    val row = cell.index / 9
    val col = cell.index % 9
    val inset = 2.dp.toPx()
    drawRect(
        color = Color.Black,
        topLeft = Offset(col * cellSize + inset, row * cellSize + inset),
        size = Size(cellSize - inset * 2, cellSize - inset * 2),
        style = Stroke(width = 1.dp.toPx())
    )
}
```
Do NOT draw error indicator on the selected cell (it would draw over the solid black fill).

**DrawScope extension for pencil marks (3×3 mini-grid convention):**
```kotlin
private fun DrawScope.drawPencilMarks(
    marks: Set<Int>,
    cellLeft: Float,
    cellTop: Float,
    cellSize: Float,
    textMeasurer: TextMeasurer,
    style: TextStyle
) {
    val subSize = cellSize / 3f
    for (digit in 1..9) {
        if (digit !in marks) continue
        val subRow = (digit - 1) / 3
        val subCol = (digit - 1) % 3
        val measured = textMeasurer.measure(digit.toString(), style)
        val x = cellLeft + subCol * subSize + (subSize - measured.size.width) / 2f
        val y = cellTop + subRow * subSize + (subSize - measured.size.height) / 2f
        drawText(textLayoutResult = measured, topLeft = Offset(x, y))
    }
}
```

**DrawScope extension for grid lines:**
```kotlin
private fun DrawScope.drawGridLines(cellSize: Float) {
    val thinStroke = 1.dp.toPx()
    val thickStroke = 2.5.dp.toPx()
    for (i in 0..9) {
        val pos = i * cellSize
        val stroke = if (i % 3 == 0) thickStroke else thinStroke
        drawLine(Color.Black, Offset(pos, 0f), Offset(pos, 9 * cellSize), stroke)
        drawLine(Color.Black, Offset(0f, pos), Offset(9 * cellSize, pos), stroke)
    }
}
```

```kotlin
package com.mudita.sudoku.ui.game

import androidx.compose.foundation.layout.*
import androidx.compose.runtime.Composable
import androidx.compose.ui.Modifier
import androidx.compose.ui.unit.dp
import com.mudita.mmd.ButtonMMD
import com.mudita.mmd.TextMMD

@Composable
fun NumberPad(
    onDigitClick: (Int) -> Unit,
    onErase: () -> Unit,
    modifier: Modifier = Modifier
) {
    Row(
        modifier = modifier.fillMaxWidth(),
        horizontalArrangement = Arrangement.spacedBy(2.dp)
    ) {
        for (digit in 1..9) {
            ButtonMMD(
                modifier = Modifier
                    .weight(1f)
                    .sizeIn(minHeight = 56.dp),
                onClick = { onDigitClick(digit) }
            ) {
                TextMMD(text = digit.toString())
            }
        }
        // Erase button at end of row — Claude's discretion (D-06 placement)
        ButtonMMD(
            modifier = Modifier
                .weight(1f)
                .sizeIn(minHeight = 56.dp),
            onClick = onErase
        ) {
            TextMMD(text = "×")
        }
    }
}
```

```kotlin
@Composable
fun ControlsRow(
    inputMode: InputMode,
    onToggleMode: () -> Unit,

    modifier: Modifier = Modifier

        containerColor = Color.Black,
```

```kotlin
package com.mudita.sudoku.ui.game

import androidx.compose.foundation.layout.*
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.unit.dp
import androidx.lifecycle.compose.collectAsStateWithLifecycle
import androidx.lifecycle.viewmodel.compose.viewModel
import com.mudita.mmd.TextMMD
import com.mudita.sudoku.game.GameViewModel
import com.mudita.sudoku.game.model.GameEvent
import com.mudita.sudoku.puzzle.model.Difficulty

@Composable
fun GameScreen(
    viewModel: GameViewModel = viewModel()
) {
    val uiState by viewModel.uiState.collectAsStateWithLifecycle()

    // Collect one-shot events (completion — handled in Phase 5)
    LaunchedEffect(Unit) {
        viewModel.events.collect { event ->
            when (event) {
                is GameEvent.Completed -> { /* TODO Phase 5: navigate to completion screen */ }
            }
        }
    }

    // Auto-start an Easy game on first load (no difficulty selection screen yet — Phase 6)
    // LaunchedEffect(Unit) fires exactly once — startGame() sets isLoading=true immediately,
    // preventing any re-entry. The fragile board.all{it==0} key approach is avoided here.
    LaunchedEffect(Unit) {
        viewModel.startGame(Difficulty.EASY)
    }

    if (uiState.isLoading) {
        LoadingScreen()
        return
    }

    Column(
        modifier = Modifier
            .fillMaxSize()
            .windowInsetsPadding(WindowInsets.systemBars)
            .padding(horizontal = 8.dp)
    ) {
        // 1. Difficulty label (D-07 top bar)
        DifficultyBar(difficulty = uiState.difficulty)

        // 2. Grid — fills remaining vertical space (D-07)
        GameGrid(
            modifier = Modifier
                .fillMaxWidth()
                .weight(1f),
            board = uiState.board,
            solution = uiState.solution,
            givenMask = uiState.givenMask,
            selectedCellIndex = uiState.selectedCellIndex,
            pencilMarks = uiState.pencilMarks,
            onCellClick = viewModel::selectCell
        )

        Spacer(modifier = Modifier.height(8.dp))

        // 3. Controls row — Fill/Pencil toggle + Undo (D-07, D-08)
        ControlsRow(
            inputMode = uiState.inputMode,
            onToggleMode = viewModel::toggleInputMode,
            onUndo = viewModel::undo
        )

        Spacer(modifier = Modifier.height(8.dp))

        // 4. Number pad (D-07, D-05, D-06)
        NumberPad(
            onDigitClick = viewModel::enterDigit,
            onErase = viewModel::eraseCell
        )

        Spacer(modifier = Modifier.height(8.dp))
    }
}

@Composable
private fun DifficultyBar(difficulty: Difficulty) {
    Box(
        modifier = Modifier
            .fillMaxWidth()
            .padding(vertical = 8.dp),
        contentAlignment = Alignment.Center
    ) {
        TextMMD(
            text = difficulty.name.lowercase()
                .replaceFirstChar { it.uppercase() }
        )
    }
}

@Composable
private fun LoadingScreen() {
    Box(
        modifier = Modifier.fillMaxSize(),
        contentAlignment = Alignment.Center
    ) {
        // Static text only — no spinner or animation (UI-02; spinner causes E-ink ghosting)
        TextMMD(text = "Generating puzzle…")
    }
}
```

**Auto-start note:** `LaunchedEffect(Unit)` fires exactly once per composition — `startGame()` is temporary for Phase 3 until the difficulty selection screen is built in Phase 6.

No `Animated*` composable, no `animateColorAsState`, no `rememberRipple()` anywhere (UI-02).
