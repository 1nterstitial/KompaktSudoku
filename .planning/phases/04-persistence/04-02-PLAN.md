    ```kotlin
            }

    }
    ```

    ```kotlin
    fun resumeGame() {

        pendingSavedState = null
        _showResumeDialog.value = false
        undoStack.clear()
        _uiState.value = saved
    }
    ```

    ```kotlin
    fun startNewGame() {

        startGame(Difficulty.EASY)
    }
    ```

    ```kotlin
    suspend fun saveNow() {
    ```
